#include <stdio.h>
#include <string.h>
int main()
{
   char a[100],tmp;
   scanf("%s",&a);
   int i,j,n=strlen(a);
   for(i=0;i<n-1;i++){
      for(j=i+1;j<n;j++){
         if(a[i]>a[j]){
            tmp=a[i];
            a[i]=a[j];
            a[j]=tmp;
         }
      }
   }
   printf("%s",a);
   return 0;
}

#include <stdio.h>
#include <string.h>

int main (void) {
   char string[100];
   scanf("%s",&string);
   char temp;

   int i, j;
   int n = strlen(string);

   printf("String before sorting - %s \n", string);

   for (i = 0; i < n-1; i++) {
      for (j = i+1; j < n; j++) {
         if (string[i] > string[j]) {
            temp = string[i];
            string[i] = string[j];
            string[j] = temp;
         }
      }
   }
   
   printf("String after sorting  - %s \n", string);
   return 0;
}



Output:-
sriram                                                                                                                                        
String before sorting - sriram                                                                                                                
String after sorting  - aimrrs
