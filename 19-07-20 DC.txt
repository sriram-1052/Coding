#include <stdio.h>

int main()
{
    int n,a[1001],i,j,x,f=0;
    scanf("%d",&n);
    for(i=0;i<n;i++){
        scanf("%d",&a[i]);
    }
    for(i=0;i<n;i++){
        if(a[i]%2==0){
            if(f==0){
                x=i;
                f=1;
            }
            else{
                for(j=i;j>=x;j--){
                    printf("%d ",a[j]);
                }
                f=0;
            }
        }
        else{
            if(f==0) printf("%d ",a[i]);
        }
    }
    if(f==1){
        for(j=x;j<n;j++){
            printf("%d ",a[j]);
        }
    }
    return 0;
}
