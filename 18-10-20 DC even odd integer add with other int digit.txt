#include <stdio.h>

int main()
{
    int n,a[1001],i,j,s,d;
    scanf("%d",&n);
    for(i=0;i<n;i++){
        scanf("%d",&a[i]);
    }
    for(i=0;i<n;i++){
        s=a[i];
        if(s%2==0){
            for(j=0;j<n;j++){
                if(j!=i){
                    d=a[j];
                    while(d>0){
                        if(d%10%2==0) s+=d%10;
                        d/=10;
                    }
                }
            }
        }
        else{
            for(j=0;j<n;j++){
                if(j!=i){
                    d=a[j];
                    while(d>0){
                        if(d%10%2!=0) s+=d%10;
                        d/=10;
                    }
                }
            }
        }
        printf("%d ",s);
    }
    return 0;
}
