#include <stdio.h>
#include <string.h>
void removeRepeatingAlphabets(char a[])
{
    int n=strlen(a),i,j,f=0,g;
    while(f==0){
        g=0;
        for(i=1;i<n;i++){
            if(a[i]==a[i-1]){
                g=1;
                for(j=i+1;j<n;j++){
                    a[j-2]=a[j];
                }
                n-=2;
            }
        }
        if(g==0) f=1;
    }
    a[n]='\0';
}

int main()
{
    char str[1001]; 
    scanf("%s", str);
    removeRepeatingAlphabets(str);
    printf("%s",str);
    return 0;
}

#include <stdio.h>

void removeRepeatingAlphabets(char a[])
{
    char b[1001],c[1001];
    int n=strlen(a),i,j,k=0,l=0,f,d;
    for(i=n-1;i>=0;i--){
        f=0;
        for(j=i-1;j>=0;j--){
            if(a[i]==a[j]){
                f=1;
                break;
            }
        }
        if(f==0){
            b[k]=a[i];
            k++;
        }
    }
    for(i=k-1;i>=0;i--){
        d=0;
        for(j=0;j<n;j++){
            if(b[i]==a[j]) d++;
        }
        if(d%2==1){
            c[l]=b[i];
            l++;
        }
    }
    for(i=0;i<n;i++) a[i]=c[i];
    return a;
}
