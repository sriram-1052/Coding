#include<stdio.h>
#include <stdlib.h>

int main()
{
int r,c,a[100][100],i,j,b[100][100];
scanf("%d\n%d",&r,&c);
for(i=0;i<r;i++){
    for(j=0;j<c;j++){
        scanf("%d",&a[i][j]);
    }
}
for(i=0;i<r;i++){
    for(j=0;j<c;j++){
        b[j][i]=a[i][j];
    }
}
for(i=c-1;i>=0;i--){
    for(j=0;j<r;j++){
        printf("%d ",b[i][j]);
    }
    printf("\n");
}
}
