#include <stdio.h>

int main()
{
    int n,a[1000],i,j,t,c=0;
    scanf("%d",&n);
    for(i=0;i<n;i++){
        scanf("%d",&a[i]);
    }
    for(i=0;i<n-1;i++){
        for(j=i+1;j<n;j++){
            if(a[i]>a[j]){
                t=a[i];
                a[i]=a[j];
                a[j]=t;
                c++;
            }
        }
    }
    printf("Array is sorted in %d swaps.\n",c);
    printf("First Element: %d\n",a[0]);
    printf("Last Element: %d",a[n-1]);
    return 0;
}

Output:-
3 2 1

Second element: 3
