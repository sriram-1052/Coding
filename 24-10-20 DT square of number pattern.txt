#include<stdio.h>
#include <stdlib.h>

int main()
{
int r;
scanf("%d",&r);
int c=(2*r)-1,i,j,x=r-1,y=r+1,a[10001];
for(i=0;i<c;i++){
    if(i==r-1) printf("%d",r*r);
    else printf("*");
}
printf("\n");
for(i=1;i<r-1;i++){
    for(j=0;j<c;j++){
        if(j==x-1) printf("%d",x*x);
        else{
            if(j==y-1) printf("%d",y*y);
            else printf("*");
        }
    }
    printf("\n");
    x--;
    y++;
}
a[0]=1;
for(i=c-1;i>1;i-=2){
    a[i]=y*y;
    y++;
}
for(i=0;i<c;i++){
    if(i%2==0) printf("%d",a[i]);
    else printf("*");
}
}
