#include <stdio.h>
#include <string.h>
int main()
{
   char a[100],tmp,b[100];
   scanf("%s %s",&a,&b);
   int i,j;
   int n=strlen(a);
   int k=strlen(b);
   for(i=0;i<n-1;i++){
      for(j=i+1;j<n;j++){
         a[i]=tolower(a[i]);
         a[j]=tolower(a[j]);
         if(a[i]>a[j]){
            tmp=a[i];
            a[i]=a[j];
            a[j]=tmp;
         }
      }
   }
   for(i=0;i<k-1;i++){
      for(j=i+1;j<k;j++){
         b[i]=tolower(b[i]);
         b[j]=tolower(b[j]);
         if(b[i]>b[j]){
            tmp=b[i];
            b[i]=b[j];
            b[j]=tmp;
         }
      }
   }
   int x=97,flag=0;
   if(a[0]==x){
       for(i=0;i<n;i++){
           for(j=i+1;j<n;j++){
               if(a[i]==a[j]){
                   i++;
                   break;
               }
               else{
                   if(a[i]!=x){
                       flag=1;
                       break;
                   }
                   else{
                       x++;
                   }
               }
           }
           if(flag==1){
               break;
           }
       }
       if(flag!=1){
           for(i=0;i<k;i++){
               for(j=i+1;j<k;j++){
                   if(b[i]==b[j]){
                       i++;
                       break;
                   }
                   else{
                       if(b[i]=x){
                           flag=1;
                           break;
                       }
                       else{
                           x++;
                       }
                   }
               }
               if(flag==1){
                   break;
               }
           }
       }
   }
   else{
       if(b[0]==x){
           for(i=0;i<k;i++){
               for(j=i+1;j<k;j++){
                   if(b[i]==b[j]){
                       i++;
                       break;
                   }
                   else{
                       if(b[i]=x){
                           flag=1;
                           break;
                       }
                       else{
                           x++;
                       }
                   }
               }
               if(flag==1){
                   break;
               }
           }
           if(flag!=1){
               for(i=0;i<n;i++){
                    for(j=i+1;j<n;j++){
                        if(a[i]==a[j]){
                            i++;
                            break;
                        }
                        else{
                            if(a[i]!=x){
                                flag=1;
                                break;
                            }
                            else{
                                x++;
                            }
                        }
                    }
                    if(flag==1){
                        break;
                    }
                }
            }
       }
       else{
           flag=1;
       }
   }
   if(flag==0) printf("yes");
   else printf("no");
   return 0;
}
