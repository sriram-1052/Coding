#include <stdio.h>
#include <string.h>
int main()
{
    char a[100],b[100];
    printf("Enter:");
    scanf("%s",&a);
    int n,k,i,x=0;
    n=strlen(a);
    k=n/2;
    if(n%2!=0)
    {
        for(i=0;i<n;i++)
        {
            if(i==k+1)
            {
                continue;
            }
            else
            {
                b[x]=a[i];
                x++;
            }
        }
    }
    int flag,j,y=x/2;
    for(i=0;i<y;i++)
    {
        flag=0;
        for(j=y;j<x;j++)
        {
            if(b[i]==b[j])
            {
                flag=1;
                break;
            }
            else
            {
                continue;
            }
        }
        if(flag==0)
        {
            printf("NO");
            break;
        }
    }
    if(flag==1)
    {
        printf("YES");
    }
    return 0;
}
