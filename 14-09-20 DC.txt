n=int(input())
a=0
b=1
c=0
f=0
d=[0,1]
while(c<=n):
    c=a+b
    a=b
    b=c
    d.append(c)
for i in range(len(d)-2):
    for j in range(i+1,len(d)-1):
        if(d[i]+d[j]==n):
            f=1
            x=d[i]
            y=d[j]
            break
    if(f==1): break
if(f==0): print("-1")
else: print(x,y)
