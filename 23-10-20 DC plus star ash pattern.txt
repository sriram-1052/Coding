#include <stdio.h>

int main()
{
    int c,i,j,k;
    scanf("%d",&c);
    int r=(c-1)/2,x=0,y=c/2,z=c-1;
    for(i=0;i<c;i++){
        if(i==x || i==y || i==z) printf("*");
        else printf("+");
    }
    printf("\n");
    for(i=1;i<r;i++){
        for(j=0;j<c;j++){
            if(j<=x+k || j>=z-k || (j>=y-k && j<=y+k)) printf("#");
            else printf("+");
        }
        k++;
        printf("\n");
    }
    return 0;
}
