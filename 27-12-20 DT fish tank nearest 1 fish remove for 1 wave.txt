#include<stdio.h>
#include <stdlib.h>

int main()
{
int r,c,i,j,k,x;
scanf("%d %d",&r,&c);
int a[r][c];
for(i=0;i<r;i++){
    for(j=0;j<c;j++){
        scanf("%d",&a[i][j]);
    }
}
scanf("%d",&x);
int b[x];
for(i=0;i<x;i++){
    scanf("%d",&b[i]);
}
for(k=0;k<x;k++){
    for(i=r-1;i>=r-b[k];i--){
        for(j=0;j<c;j++){
            if(a[i][j]!=0) a[i][j]--;
        }
    }
}
for(i=0;i<r;i++){
    for(j=0;j<c;j++){
        printf("%d ",a[i][j]);
    }
    printf("\n");
}
}
