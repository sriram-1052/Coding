#include<stdio.h>
#include <stdlib.h>

int main()
{
    long n,a[101],b[101],c,d,i,j,k=0,f=0;
    scanf("%ld",&n);
    for(i=0;i<n;i++){
        scanf("%ld",&a[i]);
    }
    for(i=0;i<n;i++){
        b[k]=a[i]*a[i];
        k++;
    }
    for(i=0;i<k-1;i++){
        for(j=i+1;j<k;j++){
            if(b[i]>b[j]){
                c=b[i];
                b[i]=b[j];
                b[j]=c;
            }
        }
    }
    for(i=0;i<k-2;i++){
        for(j=1;j<k-1;j++){
            d=b[i]+b[j];
            if(d==b[j+1]){
                f=1;
                break;
            }
        }
        if(f==1) break;
    }
    if(f==1) printf("Yes");
    else printf("No");
}
