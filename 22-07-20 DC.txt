#include <stdio.h>

int main()
{
    int n,a[1001]={1,1},i,j,k=0,l=2;
    scanf("%d",&n);
    if((n==1)||(n==2)){
        if(n==1) printf("1");
        else printf("1 1");
    }
    else{
        for(i=2;i<n;i++){
            k=0;
            for(j=0;j<l-1;j++){
                a[k]=a[j]+a[j+1];
                k++;
            }
            for(j=k;j>0;j--){
                a[j]=a[j-1];
            }
            a[0]=1;
            a[k+1]=1;
            k+=2;
            l=k;
        }
    }
    for(i=0;i<k;i++){
        printf("%d ",a[i]);
    }
    return 0;
}
