#include<stdio.h>
#include <stdlib.h>

int main()
{
int n,a[101][101],i,j,k,x,y;
scanf("%d",&n);
x=n;
for(i=0;i<n;i++){
    y=x;
    for(j=0;j<n;j++){
        a[i][j]=y;
        y+=n;
    }
    x++;
}
for(i=0;i<n;i++){
    if(i%2==0){
        for(j=0;j<n;j++){
            printf("%d ",a[i][j]);
        }
    }
    else{
        for(j=n-1;j>=0;j--){
            printf("%d ",a[i][j]);
        }
    }
    printf("\n");
}
}
