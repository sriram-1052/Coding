#include <stdio.h>

int main()
{
    int n,a[1000],i,j,tmp,b[1000];
    printf("Enter:");
    scanf("%d",&n);
    printf("Enter:");
    for(i=0;i<n;i++){
        scanf("%d",&a[i]);
    }
    for(i=0;i<n;i++){
        b[i]=a[i];
    }
    for(i=0;i<n-1;i++){
        for(j=i+1;j<n;j++){
            if(b[i]>b[j]){
                tmp=b[i];
                b[i]=b[j];
                b[j]=tmp;
            }
        }
    }
    for(i=0;i<n;i++){
        for(j=0;j<n;j++){
            if(b[i]==a[j]){
                printf("%d ",j+1);
                break;
            }
        }
    }
    return 0;
}
