#include <stdio.h>
#include <stdlib.h>
#include <string.h>
int main()
{
    char a[1001],t;
    scanf("%s",a);
    int n=strlen(a),i,j,c=0;
    for(i=0;i<n-1;i++){
        for(j=i+1;j<n;j++){
            if(a[i]>a[j]){
                t=a[i];
                a[i]=a[j];
                a[j]=t;
            }
        }
    }
    for(i=0;i<n;i++){
        if(a[i]==a[i+1]) i++;
        else c++;
    }
    if(c==0) printf("%d",c);
    else printf("%d",c-1);
    return 0;
}
