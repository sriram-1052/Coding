#include <stdio.h>

int main()
{
    int n,i,j,k=0,a[100],b[100],x,sum=0,l=0;
    printf("Enter:");
    scanf("%d",&n);
    for(i=2;i<=n;i++)
    {
        for(j=2;j<=i;j++)
        {
            if(i==j)
            {
                a[k]=i;
                k++;
                break;
            }
            else
            {
                if(i%j==0)
                {
                    break;
                }
            }
        }
    }
    printf("Prime number:");
    for(i=0;i<k;i++)
    {
        printf("%d ",a[i]);
    }
    printf("\n");
    for(i=0;i<k;i++)
    {
        x=a[i]%10;
        if(x==7)
        {
            b[l]=a[i];
            l++;
        }
    }
    printf("ending 7 prime no.:");
    for(i=0;i<l;i++)
    {
        printf("%d ",b[i]);
    }
    printf("\n");
    for(i=0;i<l;i++)
    {
        sum=sum+b[i];
    }
    printf("%d",sum);
    return 0;
}
