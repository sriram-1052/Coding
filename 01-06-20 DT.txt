#include <stdio.h>
int main()
{
int a[101][101],i,j,r,c,t,p,q,v,s;
scanf("%d %d %d",&r,&c,&t);
for(i=0;i<r;i++){
    for(j=0;j<c;j++){
        a[i][j]=0;
    }
}
while(t>0){
    scanf("%d %d %d %d",&p,&q,&v,&s);
    for(i=0;i<r;i++){
        for(j=0;j<c;j++){
            if((i>=p-1)&&(i<v)){
                if((j>=q-1)&&(j<s)){
                    a[i][j]+=1;
                }
            }
        }
    }
    t--;
}
for(i=0;i<r;i++){
    for(j=0;j<c;j++){
        printf("%d ",a[i][j]);
    }
    printf("\n");
}
}
