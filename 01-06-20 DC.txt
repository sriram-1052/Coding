#include <stdio.h>
#include <string.h>
int main()
{
    char a[1000],b[1000];
    scanf("%s %s",&a,&b);
    int i,c=strlen(a),d=strlen(b),n;
    if(c<d) n=c;
    else n=d;
    for(i=0;i<n;i++){
        printf("%c",a[i]);
    }
    printf(" ");

    return 0;
}
