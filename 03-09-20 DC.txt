#include<stdio.h>
#include <stdlib.h>

int main()
{
int r,c,a[51][51],i,j,f,d=0;
scanf("%d %d",&r,&c);
for(i=0;i<r;i++){
    for(j=0;j<c;j++){
        scanf("%d",&a[i][j]);
    }
}
for(j=0;j<c;j++){
    f=0;
    for(i=0;i<r-1;i++){
        if(a[i][j]<a[i+1][j]){
            f=1;
            break;
        }
    }
    if(f==0) d++;
}
printf("%d",d);
}
