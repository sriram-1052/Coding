#include <stdio.h>

int main()
{
    int m,n,a[101],b[101],i,j;
    scanf("%d %d",&m,&n);
    for(i=0;i<m;i++){
        scanf("%d",&a[i]);
    }
    for(i=0;i<n;i++){
        scanf("%d",&b[i]);
    }
    for(i=0;i<m;i++){
        for(j=0;j<n;j++){
            if(a[i]==b[j]){
                printf("%d",a[i]);
                break;
            }
        }
    }
    return 0;
}
