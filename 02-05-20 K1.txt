#include <stdio.h>
int main()
{
    int n,a[100],o[100],o1[100],e[100],e1[100],i,j=0,k=0,l=0;
    printf("Enter:");
    scanf("%d",&n);
    printf("Enter:");
    for(i=0;i<n;i++)
    {
        scanf("%d",&a[i]);
    }
    for(i=0;i<n;i++)
    {
        if(a[i]%2!=0)
        {
            o[l]=a[i];
            o1[l]=i;
            l++;
        }
        else
        {
            e[k]=a[i];
            e1[k]=i;
            k++;
        }
    }
    int tmp1,tmp2,tmp3,tmp4;
    for(i=0;i<l-1;i++)
    {
        for(j=i+1;j<l;j++)
        {
            if(o1[i]>o1[j])
            {
                tmp1=o1[i];
                o1[i]=o1[j];
                o1[j]=tmp1;
            }
            if(o[i]>o[j])
            {
                tmp3=o[i];
                o[i]=o[j];
                o[j]=tmp3;
            }
        }
    }
    for(i=0;i<k-1;i++)
    {
        for(j=i+1;j<k;j++)
        {
            if(e1[i]>e1[j])
            {
                tmp2=e1[i];
                e1[i]=e1[j];
                e1[j]=tmp2;
            }
            if(e[i]>e[j])
            {
                tmp4=e[i];
                e[i]=e[j];
                e[j]=tmp4;
            }
        }
    }
    int b[n];
    for(i=0;i<l;i++)
    {
        b[o1[i]]=o[i];
    }
    for(i=0;i<k;i++)
    {
        b[e1[i]]=e[i];
    }
    for(i=0;i<n;i++)
    {
        printf("%d ",b[i]);
    }
    return 0;
}
