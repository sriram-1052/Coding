#include<stdio.h>
#include <stdlib.h>

int main()
{
int n,x,a[101],b[101],d,i,j,k=0,y;
scanf("%d %d",&n,&x);
for(i=0;i<n;i++){
    scanf("%d",&a[i]);
}
for(i=0;i<n;i++){
    if(a[i]%2==0){
        b[k]=a[i];
        k++;
    }
}
for(i=0;i<x;i++){
    d=b[0];
    for(j=0;j<k-1;j++){
        b[j]=b[j+1];
    }
    b[k-1]=d;
    y=0;
    for(j=0;j<n;j++){
        if(a[j]%2==0){
            printf("%d ",b[y]);
            y++;
        }
        else printf("%d ",a[j]);
    }
    printf("\n");
}
}
