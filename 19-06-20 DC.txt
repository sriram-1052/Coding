#include <stdio.h>

int main()
{
    int n,a[100],i,j,k,d;
    scanf("%d",&n);
    for(i=0;i<n;i++){
        scanf("%d",&a[i]);
    }
    for(i=0;i<n;i++){
        d=a[i];
        while(d>=0){
            k=0;
            for(j=2;j<=d/2;j++){
                if(d%j==0){
                    k=1;
                    break;
                }
            }
            if(k==1) d++;
            else break;
        }
        printf("%d ",d-a[i]);
    }
    return 0;
}
