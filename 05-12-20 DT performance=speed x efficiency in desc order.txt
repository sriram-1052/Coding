n=int(input())
a=list(map(int,input().split()))
b=list(map(int,input().split()))
x=[]
for i in range(n):
    c=[]
    c.append(a[i])
    c.append(b[i])
    x.append(c)
x.sort(key=lambda x:-x[0])
x.sort(key=lambda x:-(x[0]*x[1]))
print(x)
