#include <stdio.h>

int main()
{
    long a,b[1000],i,j,t,k=0,c=0;
    scanf("%ld",&a);
    while(a>0){
        b[k]=a%10;
        k++;
        a/=10;
    }
    for(i=0;i<k-1;i++){
        for(j=i+1;j<k;j++){
            if(b[i]>b[j]){
                t=b[i];
                b[i]=b[j];
                b[j]=t;
            }
        }
    }
    for(i=0;i<k-1;i++){
        for(j=i+1;j<k;j++){
            if(b[i]==b[j]){
                continue;
            }
            else{
                c++;
                i=j-1;
                break;
            }
        }
        if(c>1){
            break;
        }
    }
    if(c==1){
        printf("Yes");
    }
    else{
        printf("No");
    }
    return 0;
}
