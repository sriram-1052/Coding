#include <stdio.h>

int main()
{
    int n;
    printf("Enter:");
    scanf("%d",&n);
    int a[n][n],i,j,z;
    printf("Enter:\n");
    for(i=0;i<n;i++)
    {
        for(j=0;j<n;j++)
        {
            scanf("%d",&a[i][j]);
        }
    }
    printf("Enter:");
    scanf("%d",&z);
    int b[100],k=0,x=n-1;
    for(i=0;i<n;i++)
    {
        for(j=0;j<n;j++)
        {
            if(i==j)
            {
                b[k]=a[i][j];
                k++;
                break;
            }
        }
    }
    for(i=0;i<n;i++)
    {
        for(j=n-1;j>=0;j--)
        {
            if(x==j)
            {
                b[k]=a[i][j];
                k++;
                x--;
                break;
            }
        }
    }
    printf("The elements are:");
    for(i=0;i<k;i++)
    {
        printf("%d ",b[i]);
    }
    printf("\n");
    printf("The required answer is:");
    for(i=0;i<k;i++)
    {
        if(i==z-1)
        {
            printf("%d",b[i]);
        }
    }
    return 0;
}
