#include <stdio.h>
#include <stdlib.h>
#include <string.h>
int main()
{
char a[1001],b,d;
scanf("%s",a);
int n=strlen(a),i,j,c=0;
for(i=0;i<n-1;i++){
    if(a[i]=='a' || a[i]=='A'){
        b='b';
        d='z';
    }
    else{
        if(a[i]=='z' || a[i]=='Z'){
            b='a';
            d='y';
        }
        else{
            b=tolower(a[i]+1);
            d=tolower(a[i]-1);
        }
    }
    for(j=i+1;j<n;j++){
        if(tolower(a[j])==b || tolower(a[j])==d) c++;
    }
}
printf("%d",c);
return 0;
}
