#include<stdio.h>
#include <stdlib.h>
#include<string.h>
int main()
{
char a[101];
scanf("%s",a);
int n=strlen(a),i,j,x,y,z,c=0;
for(i=0;i<n-2;i++){
    x=0;
    y=0;
    z=0;
    if(a[i]=='a') x=1;
    else{
        if(a[i]=='b') y=1;
        else z=1;
    }
    for(j=i+1;j<n;j++){
        if(a[j]=='a') x=1;
        else{
            if(a[j]=='b') y=1;
            else z=1;
        }
        if(x==1 && y==1 && z==1){
            c+=(n-j);
            break;
        }
    }
}
printf("%d",c);
}
