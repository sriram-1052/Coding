import re
a=input("Enter:")
a=re.findall('\d+',a)
b=list(map(int,a))
sum=0
for i in range(0,len(b)):
    while(b[i]>0):
        d=b[i]%10
        sum=sum+d
        b[i]//=10
print(sum)
