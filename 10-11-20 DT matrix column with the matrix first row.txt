#include <stdio.h>

int main()
{
    int n,a[51][51],i,j;
    scanf("%d",&n);
    for(i=0;i<n;i++){
        for(j=0;j<n;j++){
            scanf("%d",&a[i][j]);
        }
    }
    for(j=0;j<n;j++){
        for(i=0;i<n;i++) printf("%d ",a[a[0][i]-1][j]);
        printf("\n");
    }
    return 0;
}
