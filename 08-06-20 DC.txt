#include<stdio.h>
#include <stdlib.h>
#include <string.h>
int main()
{
char a[1001];
scanf("%s",a);
int n=strlen(a),i,j,f;
for(i=0;i<n;i++){
    for(j=0;j<n;j++){
        f=0;
        if(i==j) continue;
        else{
            if(a[i]==a[j]){
                f=1;
                break;
            }
        }
    }
    if(f==0){
        printf("%c ",a[i]);
    }
}
}

Output:-
Japan
J p n
