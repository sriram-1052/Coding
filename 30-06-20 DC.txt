#include <stdio.h>

int main()
{
    int n,a[100][100],i,j,f=0,g=0,x=0,y=0,z;
    scanf("%d",&n);
    for(i=0;i<n;i++){
        for(j=0;j<n;j++){
            scanf("%d",&a[i][j]);
        }
    }
    z=n-1;
    while(f==0){
        for(i=0;i<z;i++){
            printf("* ");
        }
        for(i=x,j=y;(i>=0)&&(j<=n-1);i--,j++){
            printf("%d ",a[i][j]);
        }
        printf("\n");
        if(g==0) z--;
        else z++;
        if(z==0) g=1;
        if(x==n-1){
            if(y==n-1) f=1;
            else y++;
        }
        else x++;
    }
    return 0;
}
