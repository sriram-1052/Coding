#include <stdio.h>
int main()
{
    int r,c,a[1001][1001],i,j;
    scanf("%d %d",&r,&c);
    for(i=0;i<r;i++){
        for(j=0;j<c;j++){
            scanf("%d",&a[i][j]);
        }
    }
    char b;
    scanf(" %c",&b);
    b=tolower(b);
    switch(b){
        case 'u':{
            for(i=0;i<r;i++){
                for(j=0;j<c;j++){
                    printf("%d ",a[0][j]);
                }
                printf("\n");
            }
        }
        case 'd':{
            for(i=0;i<r;i++){
                for(j=0;j<c;j++){
                    printf("%d ",a[r-1][j]);
                }
                printf("\n");
            }
            break;
        }
        case 'l':{
            for(i=0;i<r;i++){
                for(j=0;j<c;j++){
                    printf("%d ",a[i][0]);
                }
                printf("\n");
            }
            break;
        }
        case 'r':{
            for(i=0;i<r;i++){
                for(j=0;j<c;j++){
                    printf("%d ",a[i][c-1]);
                }
                printf("\n");
            }
            break;
        }
        default:{
            for(i=0;i<r;i++){
                for(j=0;j<c;j++){
                    printf("%d ",a[i][j]);
                }
                printf("\n");
            }
            break;
        }
    }
    return 0;
}
