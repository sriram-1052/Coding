#include <stdio.h>

int main()
{
    int n,a[1001],i,x=-1,f=0;
    scanf("%d",&n);
    for(i=0;i<n;i++){
        scanf("%d",&a[i]);
    }
    while(f==0){
        x++;
        for(i=0;i<n;i++){
            if((a[i]/10)%10==x){
                f=1;
                printf("%d ",a[i]);
            }
        }
    }
    return 0;
}
