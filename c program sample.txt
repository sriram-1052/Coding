#include <stdio.h>

int main()
{
    int n,a[101],i,j;
    scanf("%d",&n);
    for(i=0;i<n;i++){
        scanf("%d",&a[i]);
    }
    for(i=1;i<n;i++){
        if(a[i]==a[i-1]){
            for(j=i+1;j<n;j++){
                a[j-1]=a[j];
            }
            n--;
            i--;
        }
    }
    printf("%d ",n);
    for(i=0;i<n;i++){
        printf("%d ",a[i]);
    }
    return 0;
}
