#include <stdio.h>

int main()
{
    int a,b,c,d,e,f,g,x,y;
    scanf("%d:%d\n %d:%d\n %d:%d",&a,&b,&c,&d,&e,&f);
    if(b>f){
        b=60-b;
        y=b+f;
        g=1;
    }
    else{
        y=f-b;
        g=0;
    }
    if(a>=e){
        if(a==e && g==0) x=e-a;
        else{
            a=24-a;
            x=a+e;
            if(g==1) x--;
        }
    }
    else{
        x=e-a;
        if(g==1) x--;
    }
    y+=d;
    if(y>=60){
        y-=60;
        x++;
    }
    x+=c;
    if(x>=24) x-=24;
    printf("%02d:%02d",x,y);
    return 0;
}
