#include<stdio.h>
#include <stdlib.h>
#include <string.h>
int main()
{
char a[101],b[101],c[101];
scanf("%s\n %s",a,b);
int m=strlen(a),n=strlen(b),i,j,k=0,f;
for(i=0;i<m;i++){
    f=0;
    for(j=i+1;j<m;j++){
        if(a[i]==a[j]){
            f=1;
            break;
        }
    }
    if(f==0){
        c[k]=a[i];
        k++;
    }
}
for(i=0;i<k;i++){
    for(j=0;j<n;j++){
        if(c[i]==b[j]){
            printf("%c",c[i]);
        }
    }
}
}

shakes
