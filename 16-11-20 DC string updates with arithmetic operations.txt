#include <stdio.h>
#include <stdlib.h>
#include <string.h>
int main()
{
    char a[101],b[101],c;
    scanf("%s\n %s\n %c",a,b,&c);
    int m=strlen(a),n=strlen(b),i,j,k,f,g=0;
    if(c=='+') printf("%s%s",a,b);
    else{
        if(c=='-'){
            for(i=0;i<m;i++){
                f=0;
                for(j=0;j<n;j++){
                    if(a[i]==b[j]){
                        f=1;
                        for(k=j+1;k<n;k++) b[k-1]=b[k];
                        n--;
                        break;
                    }
                }
                if(f==0){
                    g=1;
                    printf("%c",a[i]);
                }
            }
            if(g==0) printf("-1");
        }
        else{
            if(m>n){
                for(i=0;i<n;i++) printf("%c%c",a[i],b[i]);
                for(i=n;i<m;i++) printf("%c",a[i]);
            }
            else{
                for(i=0;i<m;i++) printf("%c%c",a[i],b[i]);
                for(i=m;i<n;i++) printf("%c",b[i]);
            }
        }
    }
    return 0;
}
