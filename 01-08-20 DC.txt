#include <stdio.h>

int main()
{
    int n,x,a[51][51],i,j,f=0;
    scanf("%d %d",&n,&x);
    for(i=0;i<n;i++){
        for(j=0;j<n;j++){
            scanf("%d",&a[i][j]);
        }
    }
    int y=n/2,z=0;
    while(z<=n/2){
        if((a[z][y-z]!=x)||(a[z][y+z]!=x)){
            f=1;
            break;
        }
        z++;
    }
    if(f==0){
        for(i=0;i<n;i++){
            if(a[z-1][i]!=x){
                f=1;
                break;
            }
        }
    }
    if(f==0) printf("YES");
    else printf("NO");
    return 0;
}
