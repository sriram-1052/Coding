#include <stdio.h>

int main()
{
    int n,i,j,x;
    scanf("%d",&n);
    x=n/2;
    for(i=0;i<n;i++){
        if(i==n/2){
            for(j=0;j<n;j++){
                printf("*");
            }
        }
        else{
            for(j=0;j<x;j++){
                printf("-");
            }
            printf("*");
        }
        printf("\n");
        if(i<n/2) x++;
        else x--;
    }
    return 0;
}
