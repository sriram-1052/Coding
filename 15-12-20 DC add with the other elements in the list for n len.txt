n,x=map(int,input().split())
a,k=[],0
for i in range(1,x+1): a.append(i)
while(len(a)<n):
    b=[]
    for i in range(len(a)):
        if(i!=k):
            d=a[i]+a[k]
            b.append(d)
    a+=b
    k+=1
for i in range(n): print(a[i],end=" ")
