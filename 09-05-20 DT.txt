#include<stdio.h>
#include <stdlib.h>

int main()
{
int n,a,b,c,i,x=2;
scanf("%d %d %d %d",&n,&a,&b,&c);
for(i=0;i<n;i++){
    while(x>0){
        if((x%a==0)||(x%b==0)){
            if(x%c!=0){
                printf("%d ",x);
                x++;
                break;
            }
        }
        x++;
    }
}
}
