#include<stdio.h>
#include <stdlib.h>

int main()
{
int r,c,a[101][101],i,j,k,l,t,f=0;
scanf("%d %d\n",&r,&c);
for(i=0;i<r;i++){
    for(j=0;j<c;j++){
        scanf("%d",&a[i][j]);
    }
}
for(i=0,k=r-1;i<r/2;i++,k--){
    if(f==0){
        for(j=0,l=c/2;j<c/2;j++,l++){
            t=a[i][j];
            a[i][j]=a[k][l];
            a[k][l]=t;
        }
    }
    else{
        for(j=c/2,l=0;l<c/2;j++,l++){
            t=a[i][j];
            a[i][j]=a[k][l];
            a[k][l]=t;
        }
    }
    if(f==0) f=1;
    else f=0;
}
for(i=0;i<r;i++){
    for(j=0;j<c;j++){
        printf("%d ",a[i][j]);
    }
    printf("\n");
}
}
