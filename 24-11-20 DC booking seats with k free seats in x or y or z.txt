#include<stdio.h>
#include <stdlib.h>
#include<string.h>
int main()
{
    int n,x,y,z,i,j,k,f=0,c=0;
    scanf("%d %d %d %d",&n,&x,&y,&z);
    char a[n];
    for(i=0;i<n;i++){
        scanf(" %c",&a[i]);
    }
    scanf("%d",&k);
    for(i=0;i<x;i++){
        if(a[i]=='F') c++;
        else c=0;
        if(c==k){
            f=1;
            printf("Yes");
            break;
        }
    }
    if(f==0){
        y+=x;
        c=0;
        for(i=x;i<y;i++){
            if(a[i]=='F') c++;
            else c=0;
            if(c==k){
                f=1;
                printf("Yes");
                break;
            }
        }
    }
    if(f==0){
        z+=y;
        c=0;
        for(i=y;i<z;i++){
            if(a[i]=='F') c++;
            else c=0;
            if(c==k){
                f=1;
                printf("Yes");
                break;
            }
        }
    }
    if(f==0) printf("No");
    return 0;
}
