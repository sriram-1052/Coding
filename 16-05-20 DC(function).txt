#include <stdio.h>
void multiplesToBeginning(int a[],int n){
    int i;
    for(i=0;i<n;i++){
        if(a[i]%10==0){
            printf("%d ",a[i]);
        }
    }
    for(i=0;i<n;i++){
        if(a[i]%10!=0){
            printf("%d ",a[i]);
        }
    }
}
int main()
{
    int a[100],n,i;
    scanf("%d",&n);
    for(i=0;i<n;i++){
        scanf("%d",&a[i]);
    }
    multiplesToBeginning(a,n);
    return 0;
}


Output:-

10 85 41 50 30                                                                                                                                
10 50 30 85 41
