#include <stdio.h>

int main()
{
    int r,c,a[51][51],i,j;
    scanf("%d %d",&r,&c);
    for(i=0;i<r;i++){
        for(j=0;j<c;j++){
            scanf("%d",&a[i][j]);
        }
    }
    for(i=0;i<r/2;i++){
        for(j=0;j<c/2;j++){
            printf("%d ",a[i][j]);
        }
        for(j=(c/2)-1;j>=0;j--){
            printf("%d ",a[i][j]);
        }
        printf("\n");
    }
    for(i=(r/2)-1;i>=0;i--){
        for(j=0;j<c/2;j++){
            printf("%d ",a[i][j]);
        }
        for(j=(c/2)-1;j>=0;j--){
            printf("%d ",a[i][j]);
        }
        printf("\n");
    }
    return 0;
}
