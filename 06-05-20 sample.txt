#include<stdio.h>
#include <stdlib.h>

int main()
{
    int a,b,n,i;
    scanf("%d %d",&a,&b);
    n=b-a;
    int x[100],e[100],j=0,k=0,y;
    for(i=b;i>=a;i--){
        if(i%2==0){
            e[k]=i;
            k++;
        }
    }
    y=a;
    for(i=0;i<=n;i++){
        if(y%2==0){
            if(j==k){
                break;
            }
            else{
                x[i]=e[j];
                j++;
            }
        }
        else{
            x[i]=y;
        }
        y++;
    }
    for(i=0;i<=n;i++){
        printf("%d ",x[i]);
    }
}
