#include<stdio.h>
#include <stdlib.h>
int main()
{
int r,c,a[51][51],i,j,x,y,s=0;
scanf("%d %d",&r,&c);
for(i=0;i<r;i++){
    for(j=0;j<c;j++){
        scanf("%d",&a[i][j]);
    }
}
scanf("%d %d",&x,&y);
while(x>0 && y>0){
    x--;
    y--;
}
for(i=x,j=y;i<r && j<c;i++,j++) s+=a[i][j];
printf("%d",s);
}
