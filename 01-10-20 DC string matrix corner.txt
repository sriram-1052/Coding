#include <stdio.h>
#include <string.h>
int main()
{
    int r,c,i,j,k=0,l,n;
    scanf("%d %d",&r,&c);
    char a[r][c],b[1001];
    for(i=0;i<r;i++){
        for(j=0;j<c;j++){
            scanf(" %c",&a[i][j]);
        }
    }
    scanf(" %s",b);
    n=strlen(b);
    l=c-1;
    for(i=0;i<r;i++){
        if(i==0){
            for(j=0;j<c;j++){
                a[i][j]=b[k];
                k++;
                if(k==n) break;
            }
        }
        else{
            a[i][l]=b[k];
            k++;
        }
        if(k==n) break;
    }
    if(k<n){
        l=0;
        for(i=r-1;i>0;i--){
            if(i==r-1){
                for(j=c-1;j>=0;j--){
                    a[i][j]=b[k];
                    k++;
                    if(k==n) break;
                }
            }
            else{
                a[i][l]=b[k];
                k++;
            }
            if(k==n) break;
        }
    }
    for(i=0;i<r;i++){
        for(j=0;j<c;j++){
            printf("%c ",a[i][j]);
        }
        printf("\n");
    }
    return 0;
}
