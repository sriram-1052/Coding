


Sample:

#include <stdio.h>
int main()
{
    int n,a[101][101],i,j;
    scanf("%d",&n);
    for(i=0;i<n;i++){
        for(j=0;j<n;j++){
            scanf("%d",&a[i][j]);
        }
    }
    for(i=0;i<n;i++){
        for(j=n-1;j>=i;j--){
            printf("%d ",a[i][j]);
        }
        for(j=i+1;j<n;j++){
            printf("%d ",a[j][i]);
        }
        printf("\n");
    }
    return 0;
}
