#include <stdio.h>

int main()
{
    int n,i,j,k;
    scanf("%d",&n);
    k=n;
    for(j=0;j<n;j++){
        printf("(");
        for(i=1;i<=n;i++){
            printf("%d",i);
            if(i%k==0){
                printf(")");
                if(i+1<=n) printf("(");
            }
            else{
                if(i!=n) printf(" ");
            }
        }
        if(n%k!=0) printf(")");
        printf("\n");
        k--;
    }
    return 0;
}
