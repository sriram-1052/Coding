#include <stdio.h>

int main()
{
    long long n,i,d,f,s=0;
    printf("Enter:");
    scanf("%lld",&n);
    while(n>0){
        d=n%10;
        f=1;
        for(i=d;i>0;i--){
            f*=i;
        }
        s+=f;
        n/=10;
    }
    printf("%lld",s);
    return 0;
}
