#include<stdio.h>
#include <stdlib.h>
#include<string.h>
int main()
{
char a[101];
scanf("%s",a);
int n=strlen(a),i,u=0,l=0,d=0,s=0;
if(n>=8){
    for(i=0;i<n;i++){
        if(isalpha(a[i])){
            if(isupper(a[i])) u=1;
            else l=1;
        }
        else{
            if(isdigit(a[i])) d=1;
            else s=1;
        }
    }
    if(u==1 && l==1 && d==1 && s==1) printf("STRONG");
    else{
        if(u==1 && l==1){
            if(d==1 || s==1) printf("GOOD");
            else printf("MEDIUM");
        }
        else{
            if(d==1 && s==1) printf("GOOD");
            else{
                if(d==1 || s==1) printf("MEDIUM");
                else printf("WEAK");
            }
        }
    }
}
else{
    printf("WEAK");
}
}
