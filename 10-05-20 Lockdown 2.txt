#include <stdio.h>
int main()
{
    int a,b,i,j,c,f=0;
    printf("Enter:");
    scanf("%d %d",&a,&b);
    for(i=a;i<=b;i++){
        c=0;
        for(j=1;j<=i;j++){
            if(i%j==0){
                c++;
            }
            else{
                continue;
            }
        }
        if(c==3){
            printf("%d ",i);
            f=1;
        }
    }
    if(f==0) printf("-1");
    return 0;
}
