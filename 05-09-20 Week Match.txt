#include<stdio.h>
#include <stdlib.h>
#include <string.h>
int main()
{
char a[10];
scanf("%s",a);
int n,x;
scanf("%d",&n);
n--;
while(n>7) n-=7;
if(a[0]=='M') x=1;
else{
    if(a[0]=='T' && a[1]=='U') x=2;
    else{
        if(a[0]=='W') x=3;
        else{
            if(a[0]=='T')x=4;
            else{
                if(a[0]=='F') x=5;
                else{
                    if(a[0]=='S' && a[1]=='A') x=6;
                    else x=7;
                }
            }
        }
    }
}
n+=x;
while(n>7) n-=7;
switch(n){
    case 1:{
        printf("MON");
        break;
    }
    case 2:{
        printf("TUE");
        break;
    }
    case 3:{
        printf("WED");
        break;
    }
    case 4:{
        printf("THU");
        break;
    }
    case 5:{
        printf("FRI");
        break;
    }
    case 6:{
        printf("SAT");
        break;
    }
    default:{
        printf("SUN");
    }
}
}
