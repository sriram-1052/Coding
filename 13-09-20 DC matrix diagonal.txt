#include<stdio.h>
#include <stdlib.h>
#include <string.h>
int main()
{
int r,c,i,j,x=0,y=1,f=0;
scanf("%d %d",&r,&c);
char a[101][101];
for(i=0;i<r;i++){
    for(j=0;j<c;j++){
        scanf(" %c",&a[i][j]);
    }
}
while(x<r-1){
    for(i=x,j=y;(i<r-1 && j>0);i++,j--){
        if(a[i][j]>a[i+1][j-1]){
            f=1;
            break;
        }
    }
    if(f==1) break;
    if(y==c-1) x++;
    else y++;
}
if(f==0) printf("YES");
else printf("NO");
}
