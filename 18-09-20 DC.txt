#include <stdio.h>

int main()
{
    int n,x,a[101],i,j,k=0,l=0,c=0;
    scanf("%d %d",&n,&x);
    for(i=0;i<n;i++){
        scanf("%d",&a[i]);
    }
    for(i=0;i<n;i++){
        if(a[i]%2==0) k++;
        else l++;
    }
    x--;
    for(i=0;i<=l-x;i++){
        for(j=i+x;j<l;j++){
            c++;
        }
    }
    printf("%d",c*k);
    return 0;
}
