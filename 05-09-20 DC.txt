#include <stdio.h>

int main()
{
    int r,c,a[51][51],i,j,b[51],d[51],k=0,l=0,f,t;
    scanf("%d %d",&r,&c);
    for(i=0;i<r;i++){
        for(j=0;j<c;j++){
            scanf("%d",&a[i][j]);
        }
    }
    for(i=0;i<r;i++){
        b[k]=a[i][c-1];
        k++;
    }
    for(i=0;i<k;i++){
        f=0;
        for(j=i+1;j<k;j++){
            if(b[i]==b[j]){
                f=1;
                break;
            }
        }
        if(f==0){
            d[l]=b[i];
            l++;
        }
    }
    for(i=0;i<l-1;i++){
        for(j=i+1;j<l;j++){
            if(d[i]>d[j]){
                t=d[i];
                d[i]=d[j];
                d[j]=t;
            }
        }
    }
    for(k=0;k<l;k++){
        for(i=0;i<r;i++){
            if(a[i][c-1]==d[k]){
                for(j=0;j<c;j++){
                    printf("%d ",a[i][j]);
                }
                printf("\n");
            }
        }
    }
    return 0;
}
