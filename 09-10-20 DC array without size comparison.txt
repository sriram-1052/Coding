#include<stdio.h>
#include <stdlib.h>

int main()
{
int a[100001],b[100001],k=0,n=0,i,c=0;
while(scanf("%d",&a[n])==1){
    n++;
}
for(i=0;i<n;i++){
    if(a[i]==0) c++;
    else{
        if(a[i]==a[i+1]){
            b[k]=a[i]+a[i];
            k++;
            c++;
            i++;
        }
        else{
            b[k]=a[i];
            k++;
        }
    }
}
for(i=0;i<k;i++){
    printf("%d ",b[i]);
}
for(i=0;i<c;i++){
    printf("0 ");
}
}
