n,k=map(int,input().split())
a=list(map(int,input().split()))[:n]
b=list(map(int,input().split()))[:n]
j=n-1
c=0
for i in range(len(a)):
    if(a[i]==b[j]):
        c+=1
    j-=1
if(c>=k): print("YES")
else: print("NO")
