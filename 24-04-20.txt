#include <stdio.h>
#include <conio.h>
void main()
{
    int a,c=0,i;
    printf("Enter:");
    scanf("%d",&a);
    int l=1;
    int r=a-1;
    for(i=0;i<a;i++)
    {
        if(l<a)
        {
            printf("(%d,%d)",l,r);
            c=c+1;
            l++;
            r--;
        }
    }
    printf("\n");
    printf("%d",c);
    getch();
}
