#include<stdio.h>
#include <stdlib.h>

int main()
{
int r,c,a[51][51],i,j;
scanf("%d %d",&r,&c);
for(i=0;i<r;i++){
    for(j=0;j<c;j++){
        scanf("%d",&a[i][j]);
    }
}
int x=0,y=c-1;
while(x<r){
    for(i=x,j=y;i<r && j<c;i++,j++) printf("%d ",a[i][j]);
    printf("\n");
    if(y==0) x++;
    else y--;
}
}
