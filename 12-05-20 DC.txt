#include <stdio.h>

int main()
{
    int n,a[100][100],i,j,s1=0,s2=0;
    printf("Enter:");
    scanf("%d",&n);
    printf("Enter:\n");
    for(i=0;i<n;i++){
        for(j=0;j<n;j++){
            scanf("%d",&a[i][j]);
        }
    }
    int k=n-1;
    for(i=0;i<n;i++){
        for(j=0;j<n;j++){
            if(i==j){
                s1+=a[i][j];
            }
        }
    }
    for(i=0;i<n;i++){
        if(k>=0){
            s2+=a[i][k];
            k--;
        }
    }
    printf("%d",s1*s2);
    return 0;
}
