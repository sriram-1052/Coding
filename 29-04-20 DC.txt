#include <stdio.h>
#include<string.h>
int main()
{
   char a[100],temp;
   printf("Enter:");
   scanf("%s",&a);
   int i,j;
   int n=strlen(a);
   for (i=0;i<n-1;i++) 
    {
        for (j=i+1;j<n;j++) 
        {
            if(a[i]>a[j]) 
            {
                temp=a[i];
                a[i]=a[j];
                a[j]=temp;
            }
        }
    }
    int count=0;
    for(i=0;i<n;i++)
    {
        if(a[i]==a[i+1])
        {
            i++;
        }
        else
        {
            count++;
        }
    }
    if(count==1)
    {
        printf("Yes");
    }
    else
    {
        printf("No");
    }
}
