#include <stdio.h>

int main()
{
    int r,c,i,j,x=1,y=1;
    scanf("%d %d",&r,&c);
    for(i=0;i<r;i++){
        if(i==r/2){
            for(j=0;j<(c/2)-3;j++) printf("-");
            printf("WELCOME");
            for(j=(c/2)+3;j<c-1;j++) printf("-");
        }
        else{
            for(j=0;j<(c/2)-y;j++) printf("-");
            for(j=0;j<x;j++) printf(".|.");
            for(j=(c/2)+y;j<c-1;j++) printf("-");
        }
        if(i<r/2){
            x+=2;
            y+=3;
        }
        else{
            x-=2;
            y-=3;
        }
        printf("\n");
    }
    return 0;
}
