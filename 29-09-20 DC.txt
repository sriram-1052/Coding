#include <stdio.h>

int main()
{
    int n,a[101],i,k=0,s=0;
    scanf("%d",&n);
    for(i=0;i<n;i++){
        scanf("%d",&a[i]);
    }
    while(k<n-1){
        if(a[k]%9==0) k+=5;
        else{
            if(a[k]%2==0) k+=3;
            else k+=2;
        }
        s++;
    }
    printf("%d",s);
    return 0;
}
