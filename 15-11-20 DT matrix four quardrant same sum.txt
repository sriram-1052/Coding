#include<stdio.h>
#include <stdlib.h>

int main()
{
int n,a[51][51],i,j,p=0,q=0,r=0,s=0;
scanf("%d",&n);
for(i=0;i<n;i++){
    for(j=0;j<n;j++){
        scanf("%d",&a[i][j]);
    }
}
for(i=0;i<n/2;i++){
    for(j=0;j<n/2;j++) p+=a[i][j];
    for(j=n/2;j<n;j++) q+=a[i][j];
}
for(i=n/2;i<n;i++){
    for(j=0;j<n/2;j++) r+=a[i][j];
    for(j=n/2;j<n;j++) s+=a[i][j];
}
if(p==q || p==r || p==s || q==r || q==s || r==s) printf("YES");
else printf("NO");
}
