#include <stdio.h>
int main()
{
    int a,i,j,flag;
    printf("Enter:");
    scanf("%d",&a);
    for(i=1;i<=a;i++)
    {
        flag=0;
        for(j=2;j<=i;j++)
        {
            if(i==j)
            {
                flag=1;
                break;
            }
            else
            {
                if(i%j==0)
                {
                    break;
                }
                else
                {
                    continue;
                }
            }
        }
        if(flag==0)
        {
            if(i%2==0) printf("%d=Even\n",i);
            else printf("%d=Odd\n",i);
        }
        else
        {
            printf("%d=Prime no.\n",i);
        }
    }
    return 0;
}



Output:-
Enter:5                                                                                                                                       
1=Odd                                                                                                                                         
2=Prime no.                                                                                                                                   
3=Prime no.                                                                                                                                   
4=Even                                                                                                                                        
5=Prime no.
