#include <stdio.h>
int main()
{
    int n,k,a[100],i,j,t;
    printf("Enter:");
    scanf("%d %d",&n,&k);
    printf("Enter:");
    for(i=0;i<n;i++)
    {
        scanf("%d",&a[i]);
    }
    for(i=0;i<n-1;i++)
    {
        for(j=i+1;j<n;j++)
        {
            if(a[i]<a[j])
            {
                t=a[i];
                a[i]=a[j];
                a[j]=t;
            }
        }
    }
    for(i=0;i<n;i++)
    {
        printf("%d ",a[i]);
    }
    j=0;
    for(i=0;i<n-1;i++)
    {
        if(a[i]==a[i+1])
        {
            continue;
        }
        else
        {
            a[j]=a[i];
            j++;
        }
    }
    printf("\n");
    for(i=0;i<j;i++)
    {
        printf("%d ",a[i]);
    }
    printf("\n%d",a[k-1]);
    return 0;
}
