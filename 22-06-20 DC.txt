#include <stdio.h>

int main()
{
    int r,c,a[100][100],i,j,f=0,x=0,y=0;
    scanf("%d %d",&r,&c);
    for(i=0;i<r;i++){
        for(j=0;j<c;j++){
            scanf("%d",&a[i][j]);
        }
    }
    while(f==0){
        for(i=x,j=y;(i>=0)&&(j<=c-1);i--,j++){
            printf("%d ",a[i][j]);
        }
        printf("\n");
        if(x==r-1){
            if(y==c-1) f=1;
            else y++;
        }
        else x++;
    }
    return 0;
}
