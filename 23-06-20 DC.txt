#include <stdio.h>

int main()
{
    int a,b,c;
    scanf("%d-%d %d",&a,&b,&c);
    while(c>12){
        b++;
        c-=12;
    }
    while(c>0){
        if(a==12){
            b++;
            a=1;
        }
        else a++;
        c--;
    }
    printf("%02d-%d",a,b);
    return 0;
}
