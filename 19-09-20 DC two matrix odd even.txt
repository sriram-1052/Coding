#include <stdio.h>

int main()
{
    int n,a[1001],b[1001],i,f=0;
    scanf("%d",&n);
    for(i=0;i<n;i++){
        scanf("%d",&a[i]);
    }
    for(i=0;i<n;i++){
        scanf("%d",&b[i]);
    }
    for(i=0;i<n;i++){
        if(f==0){
            printf("%d ",a[i]);
            if(a[i]%2==0 && b[i]%2==0){
                printf("%d ",b[i]);
                f=1;
            }
            else{
                if(a[i]%2!=0 && b[i]%2!=0){
                    printf("%d ",b[i]);
                    f=1;
                }
            }
        }
        else{
            printf("%d ",b[i]);
            if(a[i]%2==0 && b[i]%2==0){
                printf("%d ",a[i]);
                f=0;
            }
            else{
                if(a[i]%2!=0 && b[i]%2!=0){
                    printf("%d ",a[i]);
                    f=0;
                }
            }
        }
    }
    return 0;
}
