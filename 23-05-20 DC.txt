#include<stdio.h>
#include <stdlib.h>
#include<string.h>
int main()
{
    char a[1000],t;
    printf("Enter:");
    scanf("%s",&a);
    int n=strlen(a),i,f=0;
    for(i=0;i<n;i++){
        t=tolower(a[i]);
        if((t=='x')||(t=='y')){
            continue;
        }
        else{
            f=1;
            break;
        }
    }
    if(f==0){
        printf("Yes");
    }
    else{
        printf("No");
    }
    return 0;
}
