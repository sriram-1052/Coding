#include <stdio.h>

int main()
{
    long n,a[100001],i,j=1,s;
    char b[100000];
    scanf("%ld",&n);
    for(i=0;i<n;i++){
        scanf("%ld",&a[i]);
    }
    s=a[0];
    for(i=0;i<n-1;i++){
        scanf(" %c",&b[i]);
        switch(b[i]){
            case '+':{
                s+=a[j];
                break;
            }
            case '-':{
                s-=a[j];
                break;
            }
            case '*':{
                s*=a[j];
                break;
            }
            default:{
                s/=a[j];
            }
        }
        j++;
    }
    printf("%ld",s);
    return 0;
}
