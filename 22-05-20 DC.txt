#include <stdio.h>

int main()
{
    long n,s=0,m=0,d=0,h=0;
    printf("Enter:");
    scanf("%ld",&n);
    s=n%60;
    n-=s;
    while(n>0){
        if(n%3600==0){
            h++;
            n-=3600;
        }
        else{
            if(n%60==0){
                m++;
                n-=60;
                if(m==60) h++;
            }
        }
        if(h==24){
            d++;
            h=0;
        }
    }
    printf("%ldD %02ld:%02ld:%02ld",d,h,m,s);
    return 0;
}
