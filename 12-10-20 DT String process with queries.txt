#include<stdio.h>
#include <stdlib.h>
#include <string.h>
int main()
{
char a[10001],b[11];
scanf("%s",a);
int n=strlen(a),i,j,k;
scanf("%d\n",&k);
for(j=0;j<k;j++){
    scanf(" %s",&b);
    if(b[2]=='+'){
        a[n]=a[n-1];
        n++;
    }
    else{
        if(b[2]=='S'){
            for(i=n;i>0;i--){
                a[i]=a[i-1];
            }
            n++;
        }
        else{
            for(i=0;i<n;i++){
                a[i+n]=a[i];
            }
            n*=2;
        }
    }
    for(i=0;i<n;i++){
        printf("%c",a[i]);
    }
    printf("\n");
}
}
