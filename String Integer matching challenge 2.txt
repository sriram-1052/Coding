#include<string.h>

{

    for(i=0;i<n;i++){
        if(a[i]==':'){

        }
        else{
            d=a[i];

            x=(x*10)+d;
        }
    }
    g+=x;
    o=j;
    for(i=1;i<m;i++){
        scanf("%s",&a);
        n=strlen(a);
        f=0;

            }

            }
        }
    }
