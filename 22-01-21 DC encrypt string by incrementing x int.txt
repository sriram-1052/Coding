#include<stdio.h>
#include <stdlib.h>
#include<string.h>
int main()
{
char a[101];
scanf("%s",a);
int b[101],x[101],n=strlen(a),i,j,k;
for(i=0;i<n;i++){
    scanf("%d",&b[i]);
}
for(i=0;i<n;i++){
    x[i]=(int)a[i];
    if(isupper(a[i])) x[i]-=64;
    else x[i]-=96;
}
for(i=0;i<n;i++){
    for(j=0;j<=i;j++){
        x[j]+=b[i];
        if(x[j]>26) x[j]%=26;
    }
}
for(i=0;i<n;i++){
    if(isupper(a[i])) printf("%c",(char)x[i]+64);
    else printf("%c",(char)x[i]+96);
}
}
