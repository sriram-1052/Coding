#include <stdio.h>
#include <string.h>
int main()
{
    char a[1001];
    scanf("%[^\n]s",a);
    int n=strlen(a),b,i,j,c=0,f=0;
    for(i=0;i<n;i++){
        if(a[i]==' '){
            if(f==1){
                for(j=i-c;j<=i;j++){
                    printf("%c",a[j]);
                }
            }
            c=0;
            f=0;
        }
        else{
            if(isalpha(a[i])){
                b=tolower(a[i]);
                if((b=='a')||(b=='e')||(b=='i')||(b=='o')||(b=='u')){
                    f=1;
                }
            }
            c++;
        }
    }
    if(f==1){
        for(j=i-c;j<=i;j++){
            printf("%c",a[j]);
        }
    }
    return 0;
}
