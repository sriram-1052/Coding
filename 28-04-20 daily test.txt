#include<stdio.h>
#include <stdlib.h>

int main()
{
    int a,k;
    scanf("%d %d",&a,&k);
    int b[100],i,n=0,d,c[100],j=0;
    while(a>0)
    {
        d=a%10;
        b[n]=d;
        n++;
        a=a/10;
    }
    for(i=n-1;i>=0;i--)
    {
        c[j]=b[i];
        j++;
    }
    for(i=0;i<k;i++)
    {
        if(c[i]==0)
        {
            k++;
        }
        else continue;
    }
    int z[100],x=0;
    if(k==n)
    {
        printf("0");
    }
    else
    {
        for(i=k;i<n;i++)
        {
            z[x]=c[i];
            x++;
        }
    }
    for(i=0;i<x;i++)
    {
        if(x==i+1)
        {
            if(z[i]==0)
            {
                continue;
            }
            else
            {
                printf("%d",z[i]);
            }
        }
        else
        {
            printf("%d",z[i]);
        }
    }
    return 0;
}
