#include <stdio.h>

int main()
{
    int r,c,a[101][101],i,j,k,f=0;
    scanf("%d %d",&r,&c);
    for(i=0;i<r;i++){
        for(j=0;j<c;j++){
            scanf("%d",&a[i][j]);
        }
    }
    for(i=0;i<r;i++){
        for(j=0,k=c-1;j<c;j++,k--){
            if(a[i][j]!=a[i][k]){
                f=1;
                break;
            }
        }
    }
    if(f==0){
        for(i=0;i<c;i++){
            for(j=0,k=r-1;j<r;j++,k--){
                if(a[j][i]!=a[k][i]){
                    f=1;
                    break;
                }
            }
        }
    }
    if(f==0) printf("YES");
    else printf("NO");
    return 0;
}
