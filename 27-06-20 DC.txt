#include<stdio.h>
#include <stdlib.h>

int main()
{
    int n,i,j,f=0,x=0,y;
    scanf("%d",&n);
    char a[n][n],b;
    for(i=0;i<n;i++){
        for(j=0;j<n;j++){
            scanf(" %c",&a[i][j]);
        }
    }
    b=a[0][0];
    y=n-1;
    for(i=0;i<n;i++){
        for(j=0;j<n;j++){
            if((j==x)||(j==y)){
                if(a[i][j]!=b){
                    f=1;
                    break;
                }
            }
        }
        if(f==1) break;
        else{
            x++;
            y--;
        }
    }
    if(f==0) printf("YES");
    else printf("NO");
}
