#include<stdio.h>
#include <stdlib.h>

int main()
{
int n,x,y,a[10001],b[10001],c=0,d,e,s=0,i,j,k=0,l=0;
scanf("%d\n %d %d",&n,&x,&y);
while(n>0){
    a[k]=n%2;
    k++;
    n/=2;
}
while(k<y){
    a[k]=0;
    k++;
}
for(i=k-1;i>=0;i--){
    b[l]=a[i];
    l++;
}
for(i=0;i<k;i++){
    printf("%d",a[i]);
}
printf("\n");
for(i=0;i<l;i++){
    printf("%d",b[i]);
}
printf("\n");
for(i=x-1;i<y;i++){
    if(b[i]==0) b[i]=1;
    else b[i]=0;
}
for(i=l-1;i>=0;i--){
    d=1;
    for(j=0;j<c;j++){
        d*=2;
    }
    e=d*b[i];
    s+=e;
    c++;
}
for(i=0;i<l;i++){
    printf("%d",b[i]);
}
printf("\n");
printf("%d %d %d",k,l,s);
}
