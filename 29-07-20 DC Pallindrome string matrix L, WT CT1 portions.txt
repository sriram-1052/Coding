#include <stdio.h>

int main()
{
    int n,i,k,l,x=0,y,z,f;
    scanf("%d",&n);
    char a[n][n],b[100],c[100];
    for(y=0;y<n;y++){
        for(z=0;z<n;z++){
            scanf(" %c",&a[y][z]);
        }
    }
    while(x<n){
        k=0;
        l=0;
        f=0;
        y=0;
        z=x;
        while(z<n){
            b[k]=a[y][z];
            k++;
            if(y<(n-1)-x) y++;
            else z++;
        }
        for(i=k-1;i>=0;i--){
            c[l]=b[i];
            l++;
        }
        for(i=0;i<k;i++){
            if(b[i]!=c[i]){
                f=1;
                break;
            }
        }
        if(f==0){
            for(i=0;i<k;i++){
                printf("%c",b[i]);
            }
            printf("\n");
        }
        x++;
    }
    return 0;
}
