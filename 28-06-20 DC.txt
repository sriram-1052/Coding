#include<stdio.h>
#include <stdlib.h>

int main()
{
    long r,s,a[10],c=0,d=0,n=0,i,j,f=0,g;
    scanf("%ld",&s);
    r=s;
    while(s>0){
        a[n]=s%10;
        n++;
        s/=10;
    }
    for(i=n-1;i>=0;i--){
        d=(d*10)+a[i];
        g=0;
        if((d==1)||(d==0)) g=1;
        else{
            for(j=2;j<=d/2;j++){
                if(d%j==0){
                    g=1;
                    break;
                }
            }
        }
        if(g==0){
            for(j=i-1;j>=0;j--) c=(c*10)+a[j];
            if((c==1)||(c==0)) g=1;
            else{
                for(j=2;j<=c/2;j++){
                    if(c%j==0){
                        g=1;
                        break;
                    }
                }
            }
        }
        if(g==0){
            f=1;
            break;
        }
        c=0;
    }
    if(d==r) f=0;
    if(f==0) printf("-1");
    else printf("%ld %ld",d,c);
    return 0;
}
