#include<stdio.h>
#include <stdlib.h>

int main()
{
int m,n,i,j,a[51][51],b[51],c=0,d=0,p,q,s=0,t,x,y,z;
scanf("%d %d",&m,&n);
for(i=0;i<m;i++){
    for(j=0;j<n;j++){
        scanf("%d",&a[i][j]);
    }
}
scanf("%d",&x);
for(i=0;i<m;i++){
    y=0;
    z=0;
    for(j=0;j<n;j++){
        if(a[i][j]==1) y++;
        else z++;
    }
    if(y>z){
        b[c]=y;
        c++;
        d+=y;
    }
}
if(c>0){
    for(i=0;i<c-1;i++){
        for(j=i+1;j<c;j++){
            if(b[i]<b[j]){
                t=b[i];
                b[i]=b[j];
                b[j]=t;
            }
        }
    }
    p=x/d;
    if(x%d==0) printf("%d\n",p);
    else printf("%d\n",p+1);
    for(i=0;i<p;i++){
        printf("%d ",c);
    }
    if(x%d!=0){
        q=x%d;
        for(i=0;i<c;i++){
            s++;
            q-=b[i];
            if(q<=0) break;
        }
        printf("%d",s);
    }
}
else printf("-1");
return 0;
}
