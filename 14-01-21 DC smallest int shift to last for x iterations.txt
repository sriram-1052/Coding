#include <stdio.h>

int main()
{
    int n,x,i,j,d;
    scanf("%d",&n);
    int a[n];
    for(i=0;i<n;i++){
        scanf("%d",&a[i]);
    }
    scanf("%d",&x);
    for(j=0;j<x;j++){
        if(a[0]<a[1]){
            d=a[0];
            for(i=1;i<n;i++) a[i-1]=a[i];
            a[n-1]=d;
        }
        else{
            d=a[1];
            for(i=2;i<n;i++) a[i-1]=a[i];
            a[n-1]=d;
        }
    }
    for(i=0;i<n;i++){
        printf("%d ",a[i]);
    }
    return 0;
}
