#include<stdio.h>
#include <stdlib.h>

int main()
{
int m,n,a[101],i,j,x,y,z;
scanf("%d\n %d",&m,&n);
for(i=0;i<m;i++){
    a[i]=0;
}
for(i=0;i<n;i++){
    scanf("%d %d %d",&x,&y,&z);
    for(j=x-1;j<y;j++){
        a[j]+=z;
    }
}
for(i=0;i<m;i++){
    printf("%d ",a[i]);
}
}
