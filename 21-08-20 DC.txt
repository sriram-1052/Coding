#include <stdio.h>

int main()
{
    int n,a[101][101],i,j,x;
    scanf("%d",&n);
    for(i=0;i<n;i++){
        for(j=0;j<n;j++){
            scanf("%d",&a[i][j]);
        }
    }
    scanf("%d",&x);
    while(x>=360){
        x-=360;
    }
    switch(x){
        case 0:{
            for(i=0;i<n;i++){
                for(j=0;j<n;j++){
                    printf("%d ",a[i][j]);
                }
                printf("\n");
            }
            break;
        }
        case 90:{
            for(i=0;i<n;i++){
                for(j=n-1;j>=0;j--){
                    printf("%d ",a[j][i]);
                }
                printf("\n");
            }
            break;
        }
        case 180:{
            for(i=n-1;i>=0;i--){
                for(j=n-1;j>=0;j--){
                    printf("%d ",a[i][j]);
                }
                printf("\n");
            }
            break;
        }
        case 270:{
            for(i=n-1;i>=0;i--){
                for(j=0;j<n;j++){
                    printf("%d ",a[j][i]);
                }
                printf("\n");
            }
            break;
        }
        default:{
            printf("INVALID");
        }
    }
    return 0;
}
