#include<stdio.h>
#include <stdlib.h>
#include<string.h>
int main()
{
char a[100001];
scanf("%s",a);
int n=strlen(a),i,j,k,f,m;
for(i=1;i<n;i++){
    f=0;
    if(a[i]==a[0]){
        for(k=0,j=i;j<n;k++,j++){
            if(a[k]!=a[j]){
                f=1;
                break;
            }
        }
    }
    else f=1;
    if(f==0){
        m=i;
        break;
    }
}
if(f==0) printf("%d",n-m);
else printf("-1");
}
