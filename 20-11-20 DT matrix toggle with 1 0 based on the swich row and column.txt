#include<stdio.h>
#include <stdlib.h>

int main()
{
int r,c;
scanf("%d %d",&r,&c);
int a[r][c],i,j,t,x,y;
for(i=0;i<r;i++){
    for(j=0;j<c;j++){
        a[i][j]=0;
    }
}
scanf("%d",&t);
for(j=0;j<t;j++){
    scanf("%d %d\n",&x,&y);
    for(i=0;i<c;i++){
        if(a[x-1][i]==0) a[x-1][i]=1;
        else a[x-1][i]=0;
    }
    for(i=0;i<r;i++){
        if(i!=x-1){
            if(a[i][y-1]==0) a[i][y-1]=1;
            else a[i][y-1]=0;
        }
    }
}
for(i=0;i<r;i++){
    for(j=0;j<c;j++){
        printf("%d",a[i][j]);
    }
    printf("\n");
}
}
