#include<stdio.h>
#include <stdlib.h>

int main()
{
int n,i,a,f=0,x=0,y=0;
scanf("%d",&n);
for(i=0;i<n;i++){
    scanf("%d",&a);
    if(f==0){
        x+=a;
        if(a!=6) f=1;
    }
    else{
        y+=a;
        if(a!=6) f=0;
    }
}
printf("%d %d",x,y);
}
