#include <stdio.h>

int main()
{
    int n,i,j,a,b;
    scanf("%d",&n);
    for(i=1;i<=n;i++){
        a=i;
        b=n;
        for(j=0;j<=n-i;j++){
            printf("%d ",a);
            a+=b;
            b--;
        }
        printf("\n");
    }
    return 0;
}
