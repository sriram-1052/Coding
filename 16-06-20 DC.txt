#include <stdio.h>

int main()
{
    int x,y,a[51],b[51],i,j,k=0,l,n=0,t;
    scanf("%d",&x);
    while(x>0){
        a[k]=x%10;
        k++;
        x/=10;
    }
    for(i=k-1;i>=0;i--){
        b[n]=a[i];
        n++;
    }
    for(i=0;i<k;i++){
        y=k-1;
        for(j=0;j<n;j++){
            if(j!=0) printf("-");
            for(l=0;l<b[j];l++){
                printf("%d",a[y]);
            }
            y--;
        }
        t=b[0];
        for(j=1;j<n;j++){
            b[j-1]=b[j];
        }
        b[n-1]=t;
        printf("\n");
    }
    return 0;
}
