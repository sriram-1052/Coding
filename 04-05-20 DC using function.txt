#include <stdio.h>
int main()
{
    int n,i,c=0;
    printf("Enter:");
    scanf("%d",&n);
    for(i=2;i<=n/2;++i){
        if(sum(i)==1){
            if(sum(n-i)==1){
                c++;
            }
        }
    }
    printf("%d",c);
    return 0;
}
int sum(int n){
    int i,p=1;
    for(i=2;i<=n/2;++i){
        if(n%i==0){
            p=0;
            break;
        }
    }
    return p;
}
