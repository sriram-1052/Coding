#include <stdio.h>
#include <string.h>
int main()
{
    char a[1001],b,c;
    scanf("%s",a);
    int n=strlen(a),i,f=0;
    for(i=0;i<n;i++){
        b=tolower(a[i]);
        c=tolower(a[i+1]);
        if(b+1==c){
            if(f==0){
                printf("{%c",a[i]);
                f=1;
            }
            else{
                printf("%c",a[i]);
            }
        }
        else{
            if(f==0){
                printf("%c",a[i]);
            }
            else{
                f=0;
                printf("%c}",a[i]);
            }
        }
    }
    return 0;
}
