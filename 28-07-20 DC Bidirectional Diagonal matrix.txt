#include <stdio.h>

int main()
{
    int r,c,a[101][101],i,j,x,y,f=0,k=1,l=0,g,h;
    scanf("%d %d",&r,&c);
    for(i=0;i<r;i++){
        for(j=0;j<c;j++){
            scanf("%d",&a[i][j]);
        }
    }
    scanf("%d %d",&x,&y);
    while((k<=c)&&(l<r)){
        g=0;
        h=0;
        for(i=l,j=c-k;(i<r)&&(j<c);i++,j++){
            if(a[i][j]==x){
                g=1;
            }
            else{
                if(a[i][j]==y){
                    h=1;
                }
            }
            if((g==1)&&(h==1)){
                f=1;
                break;
            }
        }
        if(f==1) break;
        if(k<c) k++;
        else l++;
    }
    if(f==0){
        k=1;
        l=0;
        while((k<=c)&&(l<r)){
            g=0;
            h=0;
            for(i=l,j=k-1;(i<r)&&(j>=0);i++,j--){
                if(a[i][j]==x){
                    g=1;
                }
                else{
                    if(a[i][j]==y){
                        h=1;
                    }
                }
                if((g==1)&&(h==1)){
                    f=1;
                    break;
                }
            }
            if(f==1) break;
            if(k<c) k++;
            else l++;
        }
    }
    if(f==1) printf("YES");
    else printf("NO");
    return 0;
}
