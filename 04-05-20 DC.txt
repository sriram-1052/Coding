#include <stdio.h>
int main()
{
    int a,b[100],i,j,k=0,c=0;
    printf("Enter:");
    scanf("%d",&a);
    for(i=2;i<=a;i++){
       for(j=2;j<=i;j++){
           if(i==j){
               b[k]=i;
               k++;
           }
           else{
               if(i%j==0){
                   break;
               }
           }
       } 
    }
    for(i=0;i<k;i++){
        for(j=i+1;j<k;j++){
            if(b[i]+b[j]==a){
                c++;
                break;
            }
        }
    }
    printf("%d",c);
    return 0;
}
