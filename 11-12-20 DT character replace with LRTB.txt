#include<stdio.h>
#include <stdlib.h>
#include<string.h>
int main()
{
    int r,c,i,j,k;
    scanf("%d %d",&r,&c);
    char a[r][c];
    for(i=0;i<r;i++){
        for(j=0;j<c;j++){
            scanf(" %c",&a[i][j]);
        }
    }
    for(i=0;i<r;i++){
        for(j=c-1;j>=0;j--){
            if(a[i][j]=='L'){
                for(k=0;k<j;k++){
                    if(a[i][k]!='L' && a[i][k]!='T' && a[i][k]!='R' && a[i][k]!='B') a[i][k]='*';
                }
                break;
            }
        }
    }
    for(i=0;i<r;i++){
        for(j=0;j<c;j++){
            if(a[i][j]=='R'){
                for(k=j+1;k<c;k++){
                    if(a[i][k]!='L' && a[i][k]!='T' && a[i][k]!='R' && a[i][k]!='B') a[i][k]='*';
                }
                break;
            }
        }
    }
    for(j=0;j<c;j++){
        for(i=r-1;i>=0;i--){
            if(a[i][j]=='T'){
                for(k=0;k<i;k++){
                    if(a[k][j]!='L' && a[k][j]!='T' && a[k][j]!='R' && a[k][j]!='B') a[k][j]='*';
                }
                break;
            }
        }
    }
    for(j=0;j<c;j++){
        for(i=0;i<r;i++){
            if(a[i][j]=='B'){
                for(k=i+1;k<r;k++){
                    if(a[k][j]!='L' && a[k][j]!='T' && a[k][j]!='R' && a[k][j]!='B') a[k][j]='*';
                }
                break;
            }
        }
    }
    for(i=0;i<r;i++){
        for(j=0;j<c;j++){
            printf("%c ",a[i][j]);
        }
        printf("\n");
    }
}
