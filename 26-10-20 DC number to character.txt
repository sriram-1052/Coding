#include <stdio.h>
#include <string.h>
int main()
{
    long n,d,i,k=0;
    scanf("%ld",&n);
    char a[1001];
    while(n>0){
        d=65+(n%26);
        a[k]=(char)d;
        k++;
        n/=26;
    }
    for(i=k-1;i>=0;i--){
        printf("%c",a[i]);
    }
    return 0;
}
