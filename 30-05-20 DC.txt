#include <stdio.h>

int main()
{
    long a,b,c[10001],d[10001],i,k=0,n=0,f=0;
    scanf("%ld %ld",&a,&b);
    while(a>0){
        c[k]=a%2;
        k++;
        a/=2;
    }
    while(b>0){
        d[n]=b%2;
        n++;
        b/=2;
    }
    if(k<n){
        n=k;
    }
    for(i=0;i<n;i++){
        if(c[i]==d[i]){
            continue;
        }
        else{
            f=1;
            printf("%ld",i+1);
            break;
        }
    }
    if(f==0){
        printf("-1");
    }
    return 0;
}
