#include <stdio.h>

int main()
{
    int r,c,a[101][101],b[101][101],e[101][101],d[101][101],i,j,k,l;
    scanf("%d %d",&r,&c);
    for(i=0;i<r;i++){
        for(j=0;j<c;j++){
            scanf("%d",&a[i][j]);
        }
    }
    for(i=0;i<r;i++){
        for(j=0;j<c;j++){
            scanf("%d",&b[i][j]);
        }
    }
    for(i=0;i<r;i++){
        if(i%2==0){
            for(j=1;j<c;j+=2) e[i][j]=a[i][j];
        }
        else{
            for(j=0;j<c;j+=2) e[i][j]=a[i][j];
        }
    }
    for(i=0;i<r;i++){
        if(i%2==0){
            for(j=1;j<c;j+=2) d[i][j]=b[i][j];
        }
        else{
            for(j=0;j<c;j+=2) d[i][j]=b[i][j];
        }
    }
    for(i=0;i<r;i++){
        if(i%2==0){
            k=0;
            for(j=c-1;j>=0;j--){
                if(j%2==0){
                    e[i][k]=a[i][j];
                    k+=2;
                }
            }
        }
        else{
            l=1;
            for(j=c-1;j>=0;j--){
                if(j%2!=0){
                    e[i][l]=a[i][j];
                    l+=2;
                }
            }
        }
    }
    for(i=0;i<r;i++){
        if(i%2==0){
            k=0;
            for(j=c-1;j>=0;j--){
                if(j%2==0){
                    d[i][k]=b[i][j];
                    k+=2;
                }
            }
        }
        else{
            l=1;
            for(j=c-1;j>=0;j--){
                if(j%2!=0){
                    d[i][l]=b[i][j];
                    l+=2;
                }
            }
        }
    }
    for(i=0;i<r;i++){
        for(j=0;j<c;j++){
            printf("%d ",e[i][j]+d[i][j]);
        }
        printf("\n");
    }
    return 0;
}
