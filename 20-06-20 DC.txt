#include <stdio.h>

int main()
{
    int n,a[100],i,f,k=0;
    scanf("%d",&n);
    for(i=0;i<n;i++){
        scanf("%d",&a[i]);
    }
    if(a[0]%2==0) f=0;
    else f=1;
    for(i=0;i<n;i++){
        if(f==0){
            if(a[i]%2==0) f=1;
            else{
                k=1;
                break;
            }
        }
        else{
            if(a[i]%2!=0) f=0;
            else{
                k=1;
                break;
            }
        }
    }
    if(k==0) printf("Yes");
    else printf("No");
    return 0;
}
