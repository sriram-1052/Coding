#include <stdio.h>

int main()
{
    int n,i,j,x,f;
    scanf("%d",&n);
    printf("1\n");
    for(i=2;i<=n;i++){
        x=2;
        f=0;
        printf("1\n");
        while(x>1){
            for(j=x;j>0;j--){
                printf("%d ",j);
            }
            printf("\n");
            if(x==i) f=1;
            if(f==0) x++;
            else x--;
        }
        printf("1\n");
    }
    return 0;
}
