#include<stdio.h>
#include <stdlib.h>
#include<string.h>
int main()
{
    int r,c,p,q,s,t,i,j,f=0;
    scanf("%d %d",&r,&c);
    char a[r][c];
    for(i=0;i<r;i++){
        for(j=0;j<c;j++){
            scanf(" %c",&a[i][j]);
        }
    }
    scanf("%d %d\n %d %d",&p,&q,&s,&t);
    for(j=q-1;j<t;j++){
        if(j==q-1 && j==t-1){
            for(i=p-1;i<s;i++) printf("%c",a[i][j]);
        }
        else{
            if(j==q-1){
                for(i=p-1;i<r;i++) printf("%c",a[i][j]);
            }
            else{
                if(j==t-1){
                    if(f==0){
                        for(i=0;i<s;i++) printf("%c",a[i][j]);
                    }
                    else{
                        for(i=r-1;i>=s-1;i--) printf("%c",a[i][j]);
                    }
                }
                else{
                    if(f==0){
                        for(i=0;i<r;i++) printf("%c",a[i][j]);
                    }
                    else{
                        for(i=r-1;i>=0;i--) printf("%c",a[i][j]);
                    }
                }
            }
        }
        if(f==0) f=1;
        else f=0;
    }
}
