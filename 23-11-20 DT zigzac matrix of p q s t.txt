#include<stdio.h>
#include <stdlib.h>
#include<string.h>
int main()
{
    int r,c,p,q,s,t,i,j,f=0;
    scanf("%d %d",&r,&c);
    char a[r][c];
    for(i=0;i<r;i++){
        for(j=0;j<c;j++){
            scanf(" %c",&a[i][j]);
        }
    }
    scanf("%d %d\n %d %d",&p,&q,&s,&t);
    for(i=p-1;i<s;i++){
        if(i==p-1 && i==s-1){
            for(j=q-1;j>=t-1;j--) printf("%c",a[i][j]);
        }
        else{
            if(i==p-1){
                for(j=q-1;j>=0;j--) printf("%c",a[i][j]);
            }
            else{
                if(i==s-1){
                    if(f==0){
                        for(j=c-1;j>=t-1;j--) printf("%c",a[i][j]);
                    }
                    else{
                        for(j=0;j<t;j++) printf("%c",a[i][j]);
                    }
                }
                else{
                    if(f==0){
                        for(j=c-1;j>=0;j--) printf("%c",a[i][j]);
                    }
                    else{
                        for(j=0;j<c;j++) printf("%c",a[i][j]);
                    }
                }
            }
        }
        if(f==0) f=1;
        else f=0;
    }
}
