#include <stdio.h>

int main()
{
    int t,n,s,a[100],b[100],i,z;
    scanf("%d",&t);
    int l1=100,l0=100,g;
    while(t>0)
    {
        scanf("%d %d",&n,&s);
        for(i=0;i<n;i++)
        {
            scanf("%d",&a[i]);
        }
        for(i=0;i<n;i++)
        {
            scanf("%d",&b[i]);
        }
        for(i=0;i<n;i++)
        {
            if(b[i]==1)
            {
                if(a[i]<l1)
                {
                    l1=a[i];
                }
            }
            else
            {
               
                if(a[i]<l0)
                {
                    l0=a[i];
                }
            }
        }
        g=l0+l1;
        z=s+g;
        if(z<100)
        {
            printf("yes");
        }
        else
        {
            printf("no");
        }
        printf("\n");
        t--;
    }
    return 0;
}
