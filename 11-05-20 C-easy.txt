#include<stdio.h>
#include <stdlib.h>
#include<string.h>
int main()
{
char a[100],t;
printf("Enter:");
scanf("%s",&a);
int n=strlen(a);
int i,s1=0,s2=0,f=0,d;
for(i=0;i<n;i++){
    if(isdigit(a[i])){
        if(f==0){
            d=a[i];
            d-=48;
            s1=(s1*10)+d;
        }
        else{
            d=a[i];
            d-=48;
            s2=(s2*10)+d;
        }
    }
    else{
        a[i]=tolower(a[i]);
        t=a[i];
        f=1;
    }
}
switch(t){
    case 'a':{
        printf("%d",s1+s2);
        break;
    }
    case 's':{
        printf("%d",s1-s2);
        break;
    }
    case 'm':{
        printf("%d",s1*s2);
        break;
    }
    default:{
        printf("%d",s1/s2);
    }
}
}
