#include <stdio.h>

int main()
{
    long n,a[10001],b,f,l,i,j,k,s;
    scanf("%ld",&n);
    for(i=0;i<n;i++){
        scanf("%ld",&a[i]);
    }
    for(i=0;i<n;i++){
        b=a[i];
        l=b%10;
        while(b>0){
            f=b%10;
            b/=10;
        }
        s=0;
        k=i;
        for(j=0;j<f;j++){
            if(k==0){
                k=n-1;
                s+=a[k];
            }
            else{
                k--;
                s+=a[k];
            }
        }
        k=i;
        for(j=0;j<l;j++){
            if(k==n-1){
                k=0;
                s+=a[k];
            }
            else{
                k++;
                s+=a[k];
            }
        }
        printf("%ld ",s);
    }
    return 0;
}
