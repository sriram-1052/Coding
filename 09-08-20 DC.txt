#include <stdio.h>
#include <string.h>
int main()
{
    char a[1001],b[1001],c[1001],d;
    scanf("%s",a);
    int n=strlen(a),i,j,x=0,y=0;
    for(i=0;i<n;i++){
        d=tolower(a[i]);
        if(d=='a' || d=='e' || d=='i' || d=='o' || d=='u'){
            b[x]=a[i];
            x++;
        }
        else{
            c[y]=a[i];
            y++;
        }
    }
    for(i=0,j=y-1;i<x || j>=0;i++,j--){
        if(i<x) printf("%c",b[i]);
        if(j>=0) printf("%c",c[j]);
    }
    return 0;
}
