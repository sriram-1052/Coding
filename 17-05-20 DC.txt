#include <stdio.h>

int main()
{
    long n,a[100],i,s;
    scanf("%ld",&n);
    for(i=0;i<n;i++){
        scanf("%ld",&a[i]);
    }
    s=a[0];
    printf("%ld ",s);
    for(i=1;i<n;i++){
        if(a[i]<s){
            s=a[i];
            printf("%ld ",s);
        }
        else{
            printf("%ld ",s);
        }
    }
    return 0;
}
