#include <stdio.h>

int main()
{
    long n,a[100],b[100],i,j,k=0;
    scanf("%ld",&n);
    for(i=0;i<n;i++){
        scanf("%ld",&a[i]);
    }
    for(i=1;i<n;i++){
        b[k]=a[i]%10;
        k++;
        a[i]/=10;
    }
    b[k]=a[0]%10;
    k++;
    a[0]/=10;
    for(i=0;i<n;i++){
        a[i]=(a[i]*10)+b[i];
    }
    for(i=0;i<n;i++){
        printf("%ld ",a[i]);
    }
    return 0;
}
