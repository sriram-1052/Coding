#include<stdio.h>
#include <stdlib.h>
#include <string.h>
int main()
{
int r,c;
scanf("%d %d",&r,&c);
int a[r][c],i,j,k,n,x,y,z;
for(i=0;i<r;i++){
    for(j=0;j<c;j++){
        scanf("%d ",&a[i][j]);
    }
}
scanf("%d",&n);
for(k=0;k<n;k++){
    scanf("%d %d\n",&x,&y);
    z=r+c;
    a[x-1][y-1]-=z;
    z=r-1;
    for(i=x,j=x-2;i<r || j>=0;i++,j--){
        if(i<r) a[i][y-1]-=z;
        if(j>=0) a[j][y-1]-=z;
        z--;
    }
    z=c-1;
    for(i=y,j=y-2;i<c || j>=0;i++,j--){
        if(i<c) a[x-1][i]-=z;
        if(j>=0) a[x-1][j]-=z;
        z--;
    }
}
for(i=0;i<r;i++){
    for(j=0;j<c;j++){
        printf("%d ",a[i][j]);
    }
    printf("\n");
}
}
