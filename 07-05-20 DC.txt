#include <stdio.h>
#include <string.h>
int main()
{
    int x=0,c=0,d,n,i;
    char a[1000];
    printf("Enter:");
    scanf("%s",&a);
    n=strlen(a);
    for(i=0;i<n;i++){
        if(a[i]=='1'){
            x++;
        }
        else{
            if(x>c){
                c=x;
            }
            x=0;
        }
    }
    if(c>1) printf("%d",c);
    else printf("-1");
    return 0;
}



Output:-

4
























#include <stdio.h>
int main()
{
    int x=0,c=0,d;
    long long n;
    printf("Enter:");
    scanf("%lld",&n);
    while(n>0){
        d=n%10;
        if(d==1){
            x++;
        }
        else{
            if(x>c){
                c=x;
            }
            x=0;
        }
        n/=10;
    }
    if(c>1) printf("%d",c);
    else printf("-1");
    return 0;
}


Output:-
Enter:101101110100011                                                                                                                         
3
