#include<stdio.h>
#include <stdlib.h>
#include <string.h>
int main()
{
char a[1001];
scanf("%s",a);
int n=strlen(a),i,p=0,q=0,r=0,s=0;
for(i=0;i<n;i++){
    a[i]=tolower(a[i]);
    if(a[i]=='u') p++;
    else{
        if(a[i]=='d') q++;
        else{
            if(a[i]=='l') r++;
            else s++;
        }
    }
}
if(p==q && r==s) printf("YES");
else printf("NO");
}
