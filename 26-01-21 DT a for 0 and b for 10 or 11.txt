#include<stdio.h>
#include <stdlib.h>
#include<string.h>
int main()
{
char a[101],b[101];
scanf("%s",a);
int n=strlen(a),i,k=0,f=0;
for(i=0;i<n;i++){
    if(a[i]=='0'){
        b[k]='a';
        k++;
    }
    else{
        if(i+1<n){
            b[k]='b';
            k++;
            i++;
        }
        else{
            f=1;
            break;
        }
    }
}
if(f==0){
    for(i=0;i<k;i++){
        printf("%c",b[i]);
    }
}
else{
    printf("-1");
}
}
