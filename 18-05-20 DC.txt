#include <stdio.h>
int main()
{
    long long x,y;
    scanf("%lld %lld",&x,&y);
    long long i,k=1,z,o,f=0,g=0;
    if(y==0){
        f=1;
    }
    while(f==0){
        i=k;
        z=0;
        o=0;
        while(i>0){
            if(i%2==1){
                o++;
            }
            else{
                z++;
            }
            i/=2;
        }
        if((z<=x)||(o<=y)){
            if((z==x)&&(o==y)){
                if(k%2==1){
                    g=k;
                }
            }
        }
        else{
            f=1;
        }
        k++;
    }
    if(g>0){
        printf("%lld",g);
    }
    else{
        printf("-1");
    }
    return 0;
}


Output:-
2 3                                                                                                                                           
25
