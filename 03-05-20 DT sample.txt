#include<stdio.h>
#include <stdlib.h>
#include <string.h>
int main()
{
char a[100],t;
int i,j,n;
scanf("%s",&a);
n=strlen(a);
for(i=0;i<n-1;i++){
    for(j=i+1;j<n;j++){
        if(a[i]>a[j]){
            t=a[i];
            a[i]=a[j];
            a[j]=t;
        }
    }
}
for(i=0;i<n;i++){
    if(a[i]==1){
        j=i;
        break;
    }
}
a[j]=0;
a[0]=1;
printf("%s",a);
}
